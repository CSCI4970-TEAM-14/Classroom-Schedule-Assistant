Detailed Description

CREATE TABLE Schedule (
id int Auto increment NOT NULL, courseId varchar(), sectId int, method varchar() NOT NULL, enroll int NOT NULL, instructor varchar(), day varchar(50) NOT NULL, start varchar(), end varchar(), room int NOT NULL, FOREIGN KEY(sectionId) REFERENCES Course(sectionId), FOREIGN KEY(room) REFERENCES Classroom(Id), FOREIGN KEY(courseId) REFERENCES Course(Id), PRIMARY KEY (id));
