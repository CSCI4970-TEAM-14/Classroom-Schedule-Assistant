Detailed Description

CREATE TABLE Section ( id int() AI NOT NULL, sectionId varchar(), courseId varchar(), method varchar(), instructor varchar(), term varchar(), enroll varchar(), FOREIGN KEY(enroll) REFERENCES Classroom(seats), FOREIGN KEY(courseId) REFERENCES Course(courseId), PRIMARY KEY (Id));
